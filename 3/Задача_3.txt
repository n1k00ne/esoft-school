/**
 * Проверяет, является ли строка правильной скобочной последовательностью.
 *
 * @param s - Строка, содержащая символы скобок: (, ), {, }, [, ]
 * @returns true, если последовательность правильная, иначе false
 */
function isValid(s: string): boolean {
  // Пустая строка считается правильной
  if (s.length === 0) return true;

  // Если длина нечётная, правильная последовательность невозможна
  if (s.length % 2 !== 0) return false;

  // Стек для хранения открывающих скобок
  const stack: string[] = [];

  // Отображение закрывающей скобки на соответствующую открывающую
  const matchingBrackets: Record<string, string> = {
    ')': '(',
    ']': '[',
    '}': '{'
  };

  for (let i = 0; i < s.length; i++) {
    const char = s[i];

    // Если символ — открывающая скобка, помещаем в стек
    if (char === '(' || char === '[' || char === '{') {
      stack.push(char);
    }
    // Если символ — закрывающая скобка
    else if (char === ')' || char === ']' || char === '}') {
      // Если стек пуст — нет соответствующей открывающей скобки
      if (stack.length === 0) {
        return false;
      }

      // Извлекаем последнюю открывающую скобку
      const lastOpen = stack.pop();

      // Проверяем, соответствует ли она типу закрывающей
      if (lastOpen !== matchingBrackets[char]) {
        return false;
      }
    }
    // Любой другой символ (по условию их нет, но на всякий случай)
    else {
      return false;
    }
  }

  // После обработки всех символов стек должен быть пустым
  // Если в стеке остались незакрытые скобки — последовательность неверна
  return stack.length === 0;
}

// Примеры использования
console.log(isValid("()"));      // true
console.log(isValid("()[]{}"));  // true
console.log(isValid("(]"));      // false
console.log(isValid("([)]"));    // false
console.log(isValid("{[]}"));    // true
console.log(isValid(""));        // true
console.log(isValid("("));       // false
console.log(isValid("((()))"));  // true
console.log(isValid("((())"));   // false
